Table users [textSize: medium, color: Black, icon: users] {
  id serial [pk]
  name varchar [not null]
  email varchar [not null, unique]
  phone varchar(20)
  role varchar [not null, default: 'client']
  password_hash VARCHAR(255) [not null]
  is_active boolean [default: false]
  created_at datetime
}

  client_id int [ref: > users.id]
}

  name varchar

  duration_days int
}
